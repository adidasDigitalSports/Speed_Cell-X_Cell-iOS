Detailed Description

The ADIXCellDevice class represents a remote XCell device.

Use this class to connect to and interact with the device. You can initialize an ADIXCellDevice with an ADIBluetoothDeviceXCell instance, that holds the necessary CBPeripheral object for the bluetooth communication, and with an ADIXCellBindingLayer instance that is responsible for the communication logic with the remote device and an miCoachXCellBusinessRules instance which responsible for parsing raw binary data downloaded from the remote device.

// Manage Device Time Example
//
// This is an example for getting and setting the date and time on the XCell device.
@implementation ManageDeviceTimeExampleViewController {
ADIXCellDevice *_xcellDevice;
}
// Get device time
- (void)getDeviceTime {
if ([_xcellDevice isConnected]) {
// Use weakSelf inside the block
__weak typeof(self) weakSelf = self;
[_xcellDevice getDeviceTime:^(NSDate *dateTime) {
// Display or log acquired time
}];
}
}
// Set time to current
- (void)setDeviceTime {
if ([_xcellDevice isConnected]) {
// Use weakSelf inside the block
__weak typeof(self) weakSelf = self;
[_xcellDevice setDeviceTime:[NSDate date] success:^(BOOL success) {
// Re-enable user interaction
}];
}
}
@end
// Heart Rate Example
// This is an example for getting and displaying heart rate coming from the XCell device.
// XCell's default mode is radioHR.
//
// XCell sends heart rate readings in every second, which is 0 if it is not attached.
@implementation HeartRateExampleViewController {
ADIXCellDevice *_xcellDevice;
}
// Response is handled in a completion block, that you must provide
- (void)setupHeartRate {
if ([_xcellDevice isConnected]) {
// Use weakSelf inside the block
__weak typeof(self) weakSelf = self;
[_xcellDevice startHeartRateUpdatesWithHandler:^(ADIHeartRate *heartRateData, NSError *error) {
// Display or log heartRate
}];
}
}
@end
