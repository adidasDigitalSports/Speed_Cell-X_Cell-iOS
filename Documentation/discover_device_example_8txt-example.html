// Discover Bluetooth Device Example
// This is an example for discovering an ADIBluetoothDevice manually.
// For simplified connecting use ADIGenerealDevice class's connect method.
#import <miCoachBluetooth/miCoachBluetooth.h>
@interface DiscoverDeviceExampleViewController () <ADIBluetoothDelegate>
@end
@implementation DiscoverDeviceExampleViewController
- (IBAction)discoverDevice:(id)sender {
ADIBluetooth *bluetooth = [ADIBluetooth sharedInstance];
bluetooth.bluetoothDelegate = self;
[ADIBluetooth.sharedInstance stopDiscovery];
[ADIBluetooth.sharedInstance startDiscoveryWithFilter:[[ADIDiscoveryFilter alloc] initWithDeviceType:self.deviceType]];
}
