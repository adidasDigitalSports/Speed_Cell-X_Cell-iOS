// Create Device Instance Example
// This is an example that shows how to create a device instance from a known serial number
#import <miCoachSensorsUI/miCoachSensorsUI.h>
#import <UIKit/UIKit.h>
@interface CreateDeviceExampleViewController ()
@end
@implementation CreateDeviceExampleViewController
- (NSArray *)createInstancesFromMyKnownSerialNumbers:(NSArray*)serialsWithTypes {
// serialsWithTypes contains NSDictionary object which stores a serial number and a device class string
// Instead of the class string you can use anything to determine the corresponding device type to the serial number
NSMutableArray *devices = [NSMutableArray array];
for (NSDictionary *known in serialsWithTypes) {
NSString *serialNumber = known[@"kSerialNumber"];
NSString *classString = known[@"kClassString"];
ADIGeneralDevice *device;
if ([classString isEqualToString:NSStringFromClass([ADIXCellDevice class])]) {
device = [miCoachSensorsUI xcellDeviceWithSerialNumber:serialNumber];
}
else if ([classString isEqualToString:NSStringFromClass([ADISpeedCellDevice class])]) {
device = [miCoachSensorsUI speedcellDeviceWithSerialNumber:serialNumber];
}
[devices addObject:device];
}
return devices;
}
@end
